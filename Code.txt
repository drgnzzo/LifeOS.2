function getCatalogos() {
  const lista = _memoSheet('Lista');
  const data  = lista.getDataRange().getValues();
  const cats  = { proyectos:[], contactos:[], conceptos:[], recurrencias:[], meses:[], clientes:[], necesidades:[] };
  for (let i = 1; i < data.length; i++) {
    const r = data[i];
    if (r[0] && r[0] !== '') cats.proyectos.push(String(r[0]));
    if (r[1] && r[1] !== '') cats.contactos.push(String(r[1]));
    if (r[2] && r[2] !== '') cats.conceptos.push(String(r[2]));
    if (r[3] && r[3] !== '') cats.recurrencias.push(String(r[3]));
    if (r[4] && r[4] !== '') cats.meses.push(String(r[4]));
    if (r[5] && r[5] !== '') cats.clientes.push(String(r[5]));
    if (r[6] && r[6] !== '' && r[6] !== 'Necesidad') cats.necesidades.push(String(r[6]));
  }
  return cats;
}

function getListaEstructura() {
  const lista = _memoSheet('Lista');
  const data  = lista.getDataRange().getValues();
  if (!data || !data.length) return { headers:[], columnas:[] };
  const headers = data[0].map(String).filter(h => h.trim() !== '');
  const columnas = headers.map((header, ci) => {
    const valores = [];
    for (let r = 1; r < data.length; r++) {
      const v = data[r][ci];
      if (v !== undefined && v !== null && String(v).trim() !== '' && isNaN(Number(v)))
        valores.push(String(v));
    }
    return { header, valores, count: valores.length };
  });
  return { headers, columnas };
}

function agregarALista(colIndex, valor) {
  try {
    invalidarCache(); // v5.007: cualquier escritura tira el cache
    const lista   = _memoSheet('Lista');
    const data    = lista.getDataRange().getValues();
    const trimmed = valor.trim();
    if (!trimmed) return { ok:false, mensaje:'Valor vacío' };
    for (let r = 1; r < data.length; r++) {
      if (String(data[r][colIndex]).toLowerCase() === trimmed.toLowerCase())
        return { ok:false, mensaje:'"' + trimmed + '" ya existe' };
    }
    let targetRow = -1;
    for (let r = 1; r < data.length; r++) {
      const v = data[r][colIndex];
      if (v === null || v === undefined || String(v).trim() === '') { targetRow = r + 1; break; }
    }
    if (targetRow === -1) targetRow = data.length + 1;
    lista.getRange(targetRow, colIndex + 1).setValue(trimmed);
    return { ok:true, mensaje:'"' + trimmed + '" agregado' };
  } catch(e) { return { ok:false, mensaje:'Error: ' + e.message }; }
}

function getSaldoDia(fechaStr) {
  try {
    const raw  = _memoSheet('RAW');
    const last = raw.getLastRow();
    if (last < 2) return { valor:0, display:'Sin datos' };
    const p = fechaStr.split('-');
    const target = new Date(parseInt(p[0]), parseInt(p[1])-1, parseInt(p[2]));
    target.setHours(0,0,0,0);
    const data = raw.getRange(2, 2, last-1, 7).getValues();
    let ultimo = null;
    let ultimaFecha = null;
    for (let i = 0; i < data.length; i++) {
      const f = new Date(data[i][0]); f.setHours(0,0,0,0);
      if (f > target) continue;
      const v = data[i][6];
      if (v !== '' && v !== null && !isNaN(Number(v)) && Number(v) !== 0) {
        if (ultimaFecha === null || f >= ultimaFecha) {
          ultimaFecha = f;
          ultimo = Number(v);
        }
      }
    }
    if (ultimo === null) return { valor:0, display:'Sin datos' };
    return { valor:ultimo, display:'$ ' + ultimo.toLocaleString('es-MX',{minimumFractionDigits:2,maximumFractionDigits:2}) };
  } catch(e) { return { valor:0, display:'Error' }; }
}

function getDatosMes() {
  try {
    const ss    = _memoSS();
    const pagos = _memoSheet('Pagos');
    const raw   = _memoSheet('RAW');
    const lastP = pagos.getLastRow();
    const lastR = raw.getLastRow();

    const configData = pagos.getRange(2, 5, lastP - 1, 2).getValues();
    const ORDEN = ['Enero','Febrero','Marzo','Abril','Mayo','Junio',
                   'Julio','Agosto','Septiembre','Octubre','Noviembre','Diciembre'];
    const mesesSet  = new Set();
    const contactos = [];

    for (let i = 0; i < configData.length; i++) {
      const colE = String(configData[i][0] || '').trim();
      const colF = String(configData[i][1] || '').trim();
      if (colE === 'Mes') continue;
      if (ORDEN.includes(colE)) mesesSet.add(colE);
      if (colF && colF !== 'Contacto' && !contactos.includes(colF)) contactos.push(colF);
    }

    const meses = [...mesesSet].sort((a,b) => ORDEN.indexOf(a) - ORDEN.indexOf(b));
    if (!meses.length || !contactos.length) return { meses:[], grupos:{} };

    const rawData = raw.getRange(2, 2, lastR - 1, 7).getValues();
    const sumas = {};
    const MESES_IDX = ['Enero','Febrero','Marzo','Abril','Mayo','Junio',
                       'Julio','Agosto','Septiembre','Octubre','Noviembre','Diciembre'];

    for (let i = 0; i < rawData.length; i++) {
      const fecha = rawData[i][0];
      if (!fecha || !(fecha instanceof Date)) continue;
      const mesNombre = MESES_IDX[fecha.getMonth()];
      if (!meses.includes(mesNombre)) continue;

      const colD = String(rawData[i][2] || '').trim();
      const colE = String(rawData[i][3] || '').trim();
      const colF = String(rawData[i][4] || '').trim();

      contactos.forEach(ct => {
        if (colD === ct || colE === ct || colF === ct) {
          const leeColH = ct === 'Final' || ct === 'Inicio';
          const monto   = Number(leeColH ? rawData[i][6] : rawData[i][5]);
          if (isNaN(monto)) return;
          const key = mesNombre + '|' + ct;
          if (leeColH) {
            sumas[key] = monto;
          } else {
            sumas[key] = (sumas[key] || 0) + monto;
          }
        }
      });
    }

    const grupos = {};
    meses.forEach(mes => {
      grupos[mes] = contactos.map(ct => ({
        ente:  ct,
        monto: sumas.hasOwnProperty(mes + '|' + ct) ? sumas[mes + '|' + ct] : null
      }));
    });

    return { meses, grupos };
  } catch(e) { return { meses:[], grupos:{}, error: e.message }; }
}

function insertarEnRAW(entrada) {
  try {
    invalidarCache(); // v5.007: cualquier escritura tira el cache
    const raw  = _memoSheet('RAW');
    const p    = entrada.fecha.split('-');
    const fObj = new Date(parseInt(p[0]), parseInt(p[1])-1, parseInt(p[2]));
    const last = raw.getLastRow();
    let insertRow = last + 1;
    if (last > 1) {
      const fechas = raw.getRange(2, 2, last-1, 1).getValues();
      for (let i = fechas.length-1; i >= 0; i--) {
        if (fObj >= new Date(fechas[i][0])) { insertRow = i+3; break; }
        if (i === 0) insertRow = 2;
      }
    }
    raw.insertRowBefore(insertRow);
    raw.getRange(insertRow, 2).setValue(fObj).setNumberFormat('dd/mm/yyyy');
    raw.getRange(insertRow, 4).setValue(entrada.proyecto);
    raw.getRange(insertRow, 5).setValue(entrada.contacto);
    raw.getRange(insertRow, 6).setValue(entrada.concepto);
    raw.getRange(insertRow, 7).setValue(Number(entrada.monto));
    raw.getRange(insertRow, 9).setValue(entrada.recurrencia);
    if (entrada.clave && entrada.clave.trim()) raw.getRange(insertRow, 10).setValue(entrada.clave);
    if (entrada.necesidad && entrada.necesidad.trim()) raw.getRange(insertRow, 11).setValue(entrada.necesidad);
    return { ok:true, fila:insertRow, mensaje:'Insertado en fila ' + insertRow };
  } catch(e) { return { ok:false, mensaje:'Error: ' + e.message }; }
}

function getGastos() {
  try {
    const pagos = _memoSheet('Pagos');
    const last  = pagos.getLastRow();
    const data  = pagos.getRange(3, 11, last - 2, 6).getValues();

    const grupos = [];
    let grupoActual = null;

    for (let i = 0; i < data.length; i++) {
      const row     = data[i];
      const totalK  = row[0];
      const concepto= row[1];
      const clave   = row[2];
      const monto   = row[3];
      const pagado  = row[4];
      const fecha   = row[5];
      const filaReal= i + 3;

      if (!concepto && !clave) continue;

      if (totalK !== null && totalK !== '' && concepto) {
        grupoActual = { concepto: String(concepto), filaTotal: filaReal, items: [] };
        grupos.push(grupoActual);
      }

      if (clave && grupoActual) {
        const montoN = typeof monto === 'number' ? monto : null;
        const fechaStr = fecha instanceof Date
          ? Utilities.formatDate(fecha, Session.getScriptTimeZone(), 'dd/MM/yyyy')
          : (fecha ? String(fecha) : '');
        grupoActual.items.push({
          clave:   String(clave),
          monto:   montoN,
          pagado:  String(pagado || ''),
          fecha:   fechaStr,
          fila:    filaReal
        });
      }
    }

    return { ok: true, grupos };
  } catch(e) { return { ok: false, grupos: [], error: e.message }; }
}

function getFijos() {
  try {
    const pagos = _memoSheet('Pagos');
    const lastRow = pagos.getLastRow();
    const data = lastRow < 3 ? [] : pagos.getRange(3, 1, lastRow - 2, 3).getValues();
    const fijos = [];

    for (let i = 0; i < data.length; i++) {
      const nombre   = data[i][0];
      const montoRaw = data[i][1];
      const fechaRaw = data[i][2];

      const nombreStr = String(nombre || '').trim();
      if (nombreStr === '') continue;

      let montoN = null;
      if (typeof montoRaw === 'number' && !isNaN(montoRaw)) {
        montoN = montoRaw;
      } else if (montoRaw !== null && montoRaw !== '') {
        const limpio = String(montoRaw).replace(/[$,\s]/g, '');
        const n = parseFloat(limpio);
        if (!isNaN(n)) montoN = n;
      }

      let fechaStr = '—';
      if (fechaRaw instanceof Date && !isNaN(fechaRaw)) {
        fechaStr = Utilities.formatDate(fechaRaw, Session.getScriptTimeZone(), 'dd/MM/yyyy');
      } else if (fechaRaw && String(fechaRaw).trim() !== '') {
        fechaStr = String(fechaRaw).trim();
      }

      fijos.push({ nombre: nombreStr, monto: montoN, fecha: fechaStr, fila: i + 3 });
    }

    return fijos;
  } catch(e) {
    return [{ nombre: 'Error: ' + e.message, monto: null, fecha: '—', fila: 0 }];
  }
}

function getNecesidades(anio, mes, fecha) {
  try {
    const raw  = _memoSheet('RAW');
    const last = raw.getLastRow();
    if (last < 2) return { niveles:[], periodo:'' };
    var tzStr  = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyy-MM-dd');
    var tzParts = tzStr.split('-');
    var hoy = new Date(parseInt(tzParts[0]), parseInt(tzParts[1])-1, parseInt(tzParts[2]), 23, 59, 59);
    var anioFiltro = anio ? parseInt(anio) : hoy.getFullYear();
    var mesFiltro  = mes  ? parseInt(mes)  : null;
    var fechaEspecifica = (fecha && fecha !== 'null' && fecha !== 'undefined') ? fecha : null;

    const rawData  = raw.getRange(2, 2, last-1, 10).getValues();
    const fechas   = rawData.map(r=>[r[0]]);
    const montos   = rawData.map(r=>[r[5]]);
    const necs     = rawData.map(r=>[r[9]]);
    const conceptos= rawData.map(r=>[r[4]]);

    const sumas = {};
    let minFecha = null, maxFecha = null;

    const EXCLUIR_NEC = ['Inicio','Final','Rectificación','Bancos'];
    for (let i = 0; i < necs.length; i++) {
      const nec   = String(necs[i][0]||'').trim();
      if (!nec) continue;
      const _proy = String(rawData[i][3]||'').trim();
      const _cont = String(rawData[i][2]||'').trim();
      const _conc = String(rawData[i][4]||'').trim();
      if (EXCLUIR_NEC.includes(_proy) || EXCLUIR_NEC.includes(_cont) || EXCLUIR_NEC.includes(_conc)) continue;
      const fechaFila = fechas[i][0];
      if (!(fechaFila instanceof Date)) continue;
      var fNec = new Date(fechaFila.getFullYear(), fechaFila.getMonth(), fechaFila.getDate());
      if (fNec > hoy) continue;
      if (fNec.getFullYear() !== anioFiltro) continue;
      if (mesFiltro && (fNec.getMonth() + 1) !== mesFiltro) continue;
      if (fechaEspecifica) {
        var partes = fechaEspecifica.split('-');
        var espDia = parseInt(partes[2]);
        if (fNec.getDate() > espDia) continue;
      }
      const monto = Number(montos[i][0]);
      if (isNaN(monto)) continue;
      if (!minFecha || fechaFila < minFecha) minFecha = fechaFila;
      if (!maxFecha || fechaFila > maxFecha) maxFecha = fechaFila;
      const m = nec.match(/^(\d+)\./);
      if (!m) continue;
      const key = m[1];
      if (!sumas[key]) sumas[key] = { total:0, conceptos:[] };
      sumas[key].total += monto;
      const conc = String(conceptos[i][0]||'').trim();
      if (conc && !sumas[key].conceptos.includes(conc)) sumas[key].conceptos.push(conc);
    }

    const tz = Session.getScriptTimeZone();
    const fmtF = f => f ? Utilities.formatDate(f, tz, 'dd/MM/yy') : '';
    const periodo = minFecha ? fmtF(minFecha) + ' – ' + fmtF(maxFecha) : '';

    const niveles = ['1','2','3','4','5'].map(key => ({
      key,
      total:      sumas[key] ? sumas[key].total : 0,
      conceptos:  sumas[key] ? sumas[key].conceptos : []
    }));

    const MESES_NOMBRE = ['Enero','Febrero','Marzo','Abril','Mayo','Junio','Julio','Agosto','Septiembre','Octubre','Noviembre','Diciembre'];
    const rawPorMes = {};
    for (let i = 0; i < necs.length; i++) {
      const nec   = String(necs[i][0]||'').trim();
      if (!nec) continue;
      // v5.215 — FIX: este bucle (rawPorMes) NO aplicaba la exclusión de
      // Inicio/Final/Rectificación/Bancos que sí aplica el bucle de niveles.
      // Por eso Necesidades calculaba de más, en compacta y expandida.
      // Se añade la misma exclusión + filtro de año para coherencia.
      const _proy2 = String(rawData[i][3]||'').trim();
      const _cont2 = String(rawData[i][2]||'').trim();
      const _conc2 = String(rawData[i][4]||'').trim();
      if (EXCLUIR_NEC.includes(_proy2) || EXCLUIR_NEC.includes(_cont2) || EXCLUIR_NEC.includes(_conc2)) continue;
      const fechaFila2 = fechas[i][0];
      if (!(fechaFila2 instanceof Date) || fechaFila2 > hoy) continue;
      if (fechaFila2.getFullYear() !== anioFiltro) continue;
      const monto = Number(montos[i][0]);
      if (isNaN(monto)) continue;
      const m2 = nec.match(/^(\d+)\./);
      if (!m2) continue;
      const mesNom = MESES_NOMBRE[fechaFila2.getMonth()];
      if (!rawPorMes[mesNom]) rawPorMes[mesNom] = [];
      rawPorMes[mesNom].push({
        key:      m2[1],
        monto,
        concepto: String(conceptos[i][0]||'').trim()
      });
    }

    const ORDEN_M = ['Enero','Febrero','Marzo','Abril','Mayo','Junio',
                     'Julio','Agosto','Septiembre','Octubre','Noviembre','Diciembre'];
    const mesesDisponibles = Object.keys(rawPorMes)
      .sort((a,b) => ORDEN_M.indexOf(a) - ORDEN_M.indexOf(b));

    return { ok:true, niveles, periodo, rawPorMes, mesesDisponibles,
      _debug: {
        anioFiltro, mesFiltro, fechaParam: fecha,
        hoyAnio: hoy.getFullYear(), hoyMes: hoy.getMonth()+1, hoyDia: hoy.getDate(),
        esHoy: (anioFiltro === hoy.getFullYear() && mesFiltro === (hoy.getMonth()+1)),
        totalFilas: rawData.length,
        filasConNec: Object.values(sumas).reduce(function(s,v){ return s + (v.conceptos||[]).length; }, 0)
      }
    };
  } catch(e) { return { ok:false, niveles:[], error:e.message }; }
}

    const raw  = _memoSheet('RAW');
    const last = raw.getLastRow();
    if (last < 2) return { meses:[], grupos:{} };

    const hoy        = new Date();
    const mesHoyIdx  = hoy.getMonth();
    const anioHoy    = hoy.getFullYear();
    const data = raw.getRange(2, 2, last-1, 6).getValues();

    const ORDEN = ['Enero','Febrero','Marzo','Abril','Mayo','Junio',
                   'Julio','Agosto','Septiembre','Octubre','Noviembre','Diciembre'];
    const grupos = {};

    for (let i = 0; i < data.length; i++) {
      const fecha = data[i][0];
      if (!(fecha instanceof Date)) continue;
      if (fecha.getFullYear() > anioHoy) continue;

      const monto    = Number(data[i][5]);
      if (isNaN(monto) || monto === 0) continue;

      const proyecto = String(data[i][2]||'').trim();
      const contacto = String(data[i][3]||'').trim();
      const concepto = String(data[i][4]||'').trim();

      const excluir = ['Inicio','Final','Rectificación','Bancos'];
      const esExcluido = excluir.some(k => proyecto===k||contacto===k||concepto===k);
      if (esExcluido) continue;

      const mes = ORDEN[fecha.getMonth()];
      if (!grupos[mes]) grupos[mes] = { ingresos:0, egresos:0, excedente:0 };

      if (monto > 0) grupos[mes].ingresos += monto;
      else           grupos[mes].egresos  += monto;
    }

    Object.keys(grupos).forEach(mes => {
      grupos[mes].excedente = grupos[mes].ingresos + grupos[mes].egresos;
    });

    const meses = Object.keys(grupos).sort((a,b) => ORDEN.indexOf(a)-ORDEN.indexOf(b));
    return { ok:true, meses, grupos };
  } catch(e) { return { ok:false, meses:[], grupos:{}, error:e.message }; }
}

function setActivityCheck(tipo, fila, dia, valor) {
  try {
    invalidarCache(); // v5.007: cualquier escritura tira el cache
    var ss   = _memoSS();
    var hoja = _memoSheet('Activity Check');
    if(!hoja) return { ok:false, mensaje:'Hoja no encontrada' };
    // v6.072 — estructura nueva (28 columnas, 1-based):
    //   Libro:  check "Leído" 19 (S),  fecha "Cumplido" 20 (T)
    //   Movie:  check "Visto" 23 (W),  fecha "Cumplido" 24 (X)
    //   NoRut:  check "Hecho" 27 (AA), fecha "Cumplido" 28 (AB)
    var COL = {
      personal:    { L:2, M:3, W:4, J:5, V:6, S:7, D:8 },
      electronics: { L:12, M:13, W:14, J:15, V:16 },
      libro:       { check:19, fecha:20 },
      movie:       { check:23, fecha:24 },
      norut:       { check:27, fecha:28 }
    };
    var tipoMap = COL[tipo];
    if(!tipoMap) return { ok:false, mensaje:'Tipo desconocido: '+tipo };
    var col = tipoMap[dia] || tipoMap['check'];
    if(!col) return { ok:false, mensaje:'Día desconocido: '+dia };
    var newVal = (valor === true || valor === 'true');
    var celda = hoja.getRange(fila, col);
    celda.setValue(newVal);
    var tz = Session.getScriptTimeZone();
    var fechaIso = Utilities.formatDate(new Date(), tz, 'yyyy-MM-dd HH:mm');
    // Guardar fecha como NOTA de la celda (compatibilidad).
    if(newVal){ celda.setNote(fechaIso); } else { celda.clearNote(); }
    // v6.072: para Libro/Movie/NoRutinaria, escribir también la fecha
    // VISIBLE en su columna "Cumplido". Al desmarcar, se limpia.
    if(tipoMap.fecha){
      var celdaFecha = hoja.getRange(fila, tipoMap.fecha);
      if(newVal){
        celdaFecha.setValue(Utilities.formatDate(new Date(), tz, 'yyyy-MM-dd'));
      } else {
        celdaFecha.clearContent();
      }
    }
    return { ok:true, fila:fila, col:col, valor:newVal };
  } catch(e){ return { ok:false, mensaje:e.message }; }
}

    const fecha = datos.fecha ? new Date(datos.fecha + 'T12:00:00') : new Date();
    hoja.getRange(fila, 1).setValue(fecha).setNumberFormat('dd/mm/yyyy');
    hoja.getRange(fila, 2).setValue(datos.tipo || '');
    hoja.getRange(fila, 3).setValue(datos.descripcion || '');
    hoja.getRange(fila, 4).setValue(datos.doctor || '');
    if (datos.proxima) {
      hoja.getRange(fila, 5).setValue(new Date(datos.proxima + 'T12:00:00')).setNumberFormat('dd/mm/yyyy');
    }
    hoja.getRange(fila, 6).setValue(datos.notas || '');
    hoja.getRange(fila, 7).setValue(datos.estado || 'Pendiente');
    return { ok: true, fila, mensaje: 'Registro de salud guardado' };
  } catch(e) { return { ok: false, mensaje: e.message }; }
}

//  MÓDULO FINANCIERO AVANZADO
// ══════════════════════════════════════════════════════════
function getFinancieroAvanzado() {
  try {
    const ss   = _memoSS();
    const raw  = _memoSheet('RAW');
    const last = raw.getLastRow();
    if (last < 2) return { ok: false, mensaje: 'Sin datos' };

    const hoy        = new Date(); hoy.setHours(23,59,59,0);
    const anioHoy    = hoy.getFullYear();
    const mesHoyIdx  = hoy.getMonth();
    const diaHoy     = hoy.getDate();
    const data = raw.getRange(2, 2, last - 1, 7).getValues();

    let ingresosHoy=0, egresosHoy=0;
    let ingresosMes=0, egresosMes=0;
    let egresosMesHastaHoy=0, ingresosMesHastaHoy=0;
    let ingresosSemana=0, egresosSemana=0;
    let totalIngresos=0, totalEgresos=0;
    const gastosPorDia = {};
    let saldoActual = 0;

    const inicioSemana = new Date(hoy);
    inicioSemana.setDate(hoy.getDate() - (hoy.getDay()===0?6:hoy.getDay()-1));
    inicioSemana.setHours(0,0,0,0);

    const EXCLUIR = ['Inicio','Final','Rectificación','Bancos'];
    const hoyLimpio = new Date(anioHoy, mesHoyIdx, diaHoy);

    for (let i = 0; i < data.length; i++) {
      const fecha = data[i][0];
      if (!(fecha instanceof Date)) continue;

      const fAnio = fecha.getFullYear();
      const fMes  = fecha.getMonth();
      const fDia  = fecha.getDate();
      const fDate = new Date(fAnio, fMes, fDia);

      if (fAnio > anioHoy) continue;
      if (fAnio === anioHoy && fMes > mesHoyIdx) continue;
      const esFuturoEnMesActual = (fAnio === anioHoy && fMes === mesHoyIdx && fDate > hoyLimpio);

      const proyecto = String(data[i][2]||'').trim();
      const contacto = String(data[i][3]||'').trim();
      const concepto = String(data[i][4]||'').trim();
      const monto    = Number(data[i][5]);
      const saldo    = Number(data[i][6]);

      if (isNaN(monto) || monto === 0) continue;
      const esExcluido = EXCLUIR.some(k=>proyecto===k||contacto===k||concepto===k);
      if (!esFuturoEnMesActual && !isNaN(saldo) && saldo !== 0) saldoActual = saldo;
      if (esExcluido) continue;

      if (!esFuturoEnMesActual) {
        if (monto > 0) totalIngresos += monto;
        else           totalEgresos  += monto;
      }
      if (fAnio === anioHoy && fMes === mesHoyIdx) {
        if (monto > 0) ingresosMes += monto;
        else           egresosMes  += monto;
      }
      if (fAnio === anioHoy && fMes === mesHoyIdx && !esFuturoEnMesActual) {
        if (monto > 0) ingresosMesHastaHoy += monto;
        else           egresosMesHastaHoy  += monto;
      }
      if (!esFuturoEnMesActual && fAnio === anioHoy && fMes === mesHoyIdx && fDia === diaHoy) {
        if (monto > 0) ingresosHoy += monto;
        else           egresosHoy  += monto;
      }
      if (!esFuturoEnMesActual && fDate >= inicioSemana) {
        if (monto > 0) ingresosSemana += monto;
        else           egresosSemana  += monto;
      }
      if (!esFuturoEnMesActual && monto < 0) {
        const key = fAnio+'-'+fMes+'-'+fDia;
        gastosPorDia[key] = (gastosPorDia[key]||0) + Math.abs(monto);
      }
    }

    const hace30 = new Date(hoy); hace30.setDate(hoy.getDate()-30);
    let sumaDiaria=0, diasConGasto=0;
    Object.entries(gastosPorDia).forEach(([key,v])=>{
      const parts = key.split('-');
      const d = new Date(parseInt(parts[0]),parseInt(parts[1]),parseInt(parts[2]));
      if (d >= hace30){ sumaDiaria+=v; diasConGasto++; }
    });
    const gastoPorDiaPromedio = diasConGasto > 0 ? sumaDiaria/diasConGasto : 0;

    const diasRestantesMes = new Date(anioHoy, mesHoyIdx+1, 0).getDate() - diaHoy;
    const proyeccionEgresosMes = Math.abs(egresosMes);
    const proyeccionExcedente = ingresosMes - proyeccionEgresosMes;
    const runwayDias = gastoPorDiaPromedio > 0 ? Math.floor(saldoActual / gastoPorDiaPromedio) : null;
    const pctAhorro = ingresosMes > 0 ? ((ingresosMes + egresosMes)/ingresosMes*100) : 0;

    let gastoMitad1=0, gastoMitad2=0;
    for(let i=0;i<data.length;i++){
      const f=data[i][0]; if(!(f instanceof Date)) continue;
      if(f.getFullYear()!==anioHoy||f.getMonth()!==mesHoyIdx) continue;
      const m=Number(data[i][5]); if(isNaN(m)||m>=0) continue;
      if(f.getDate()<=15) gastoMitad1+=Math.abs(m);
      else gastoMitad2+=Math.abs(m);
    }
    const velocidadGasto = gastoMitad1 > 0
      ? ((gastoMitad2 - gastoMitad1)/gastoMitad1*100).toFixed(1)
      : null;

    return {
      ok: true,
      hoy:    { ingresos: ingresosHoy, egresos: Math.abs(egresosHoy), balance: ingresosHoy + egresosHoy },
      semana: { ingresos: ingresosSemana, egresos: Math.abs(egresosSemana), balance: ingresosSemana + egresosSemana },
      mes: {
        ingresos:  ingresosMes,
        egresos:   Math.abs(egresosMes),
        excedente: ingresosMes + egresosMes,
        proyeccion: {
          egresos:      Math.round(proyeccionEgresosMes),
          excedente:    Math.round(proyeccionExcedente),
          diasRestantes: diasRestantesMes,
        },
      },
      metricas: {
        gastoPorDiaPromedio: Math.round(gastoPorDiaPromedio),
        runwayDias:          runwayDias,
        porcentajeAhorro:    Math.round(pctAhorro * 10) / 10,
        velocidadGasto:      velocidadGasto ? parseFloat(velocidadGasto) : null,
        saldoActual:         saldoActual,
      },
    };
  } catch(e) { return { ok: false, error: e.message }; }
}

// ══════════════════════════════════════════════════════════
//  REVISIÓN SEMANAL / MENSUAL
// ══════════════════════════════════════════════════════════
function getRevision(tipo, anio, mes, semana) {
  try {
    const hoy = new Date(); hoy.setHours(23,59,59,0);
    var fin   = new Date(hoy);
    var inicio = new Date(hoy);

    var anioN   = anio   ? parseInt(anio)   : hoy.getFullYear();
    var mesN    = mes    ? parseInt(mes)     : null;
    var semanaN = semana ? parseInt(semana)  : null;

    if (mesN && semanaN) {
      var primerDia = new Date(anioN, mesN - 1, 1);
      var dow = primerDia.getDay();
      var offsetLunes = (dow === 0) ? 1 : (dow === 1 ? 0 : 8 - dow);
      var inicioSem1 = new Date(anioN, mesN - 1, 1 + offsetLunes);
      inicio = new Date(inicioSem1);
      inicio.setDate(inicioSem1.getDate() + (semanaN - 1) * 7);
      fin = new Date(inicio);
      fin.setDate(inicio.getDate() + 6);
      fin.setHours(23,59,59,0);
      if (fin > hoy) fin = new Date(hoy);
    } else if (mesN) {
      inicio = new Date(anioN, mesN - 1, 1, 0, 0, 0);
      fin    = new Date(anioN, mesN, 0, 23, 59, 59);
      if (fin > hoy && inicio > hoy) fin = new Date(hoy);
    } else if (tipo === 'semanal') {
      inicio.setDate(hoy.getDate() - 6);
      inicio.setHours(0,0,0,0);
    } else if (tipo === 'anual') {
      inicio = new Date(anioN, 0, 1, 0, 0, 0);
      fin    = new Date(anioN, 11, 31, 23, 59, 59);
      if (fin > hoy) fin = new Date(hoy);
    } else {
      inicio = new Date(anioN, hoy.getMonth(), 1, 0, 0, 0);
    }
    inicio.setHours(0,0,0,0);

    const ss  = _memoSS();
    const raw = _memoSheet('RAW');
    const last = raw.getLastRow();

    var ingresos = 0, egresos = 0;
    var nivelSumas = {};
    var EXCLUIR = ['Inicio','Final','Rectificación','Bancos'];

    if (last > 1) {
      var data = raw.getRange(2, 2, last - 1, 10).getValues();
      for (var i = 0; i < data.length; i++) {
        var fecha = data[i][0];
        if (!(fecha instanceof Date)) continue;
        var fClean = new Date(fecha.getFullYear(), fecha.getMonth(), fecha.getDate());
        if (fClean < inicio || fClean > fin) continue;

        var proyecto = String(data[i][2]||'').trim();
        var contacto = String(data[i][3]||'').trim();
        var concepto = String(data[i][4]||'').trim();
        var monto    = Number(data[i][5]);
        var nec      = String(data[i][9]||'').trim();

        var esExcluido = EXCLUIR.some(function(k){ return proyecto===k||contacto===k||concepto===k; });
        if (esExcluido || isNaN(monto) || monto === 0) continue;

        if (monto > 0) ingresos += monto;
        else           egresos  += monto;

        if (nec) {
          var nm = nec.match(/^(\d+)\./);
          if (nm) {
            var key = nm[1];
            nivelSumas[key] = (nivelSumas[key]||0) + monto;
          }
        }
      }
    }

    var excedente = ingresos + egresos;
    var pctAhorro = ingresos > 0 ? Math.round(excedente / ingresos * 100) : 0;

    var nivelDominante = null;
    var maxNivel = 0;
    ['1','2','3','4','5'].forEach(function(k){
      var abs = Math.abs(nivelSumas[k]||0);
      if (abs > maxNivel) { maxNivel = abs; nivelDominante = k; }
    });

    var logros = getLogros();
    var logrosCompletados = (logros.items||[]).filter(function(it){ return it.completado==='Sí'||it.completado==='Si'; }).length;
    var logrosPendientes  = (logros.items||[]).filter(function(it){ return it.completado!=='Sí'&&it.completado!=='Si'; }).length;

    var scoreInversionista = ingresos > 0 ? Math.min(100, Math.max(0, Math.round(excedente/ingresos*100))) : 0;
    var scoreConsumidor    = 100 - scoreInversionista;

    var insights = [];
    if (pctAhorro < 0)
      insights.push({ tipo:'alerta', msg: 'Gastos superaron ingresos en este período' });
    else if (pctAhorro < 10)
      insights.push({ tipo:'alerta', msg: 'Ahorro por debajo del 10% en este período' });
    else if (pctAhorro >= 20)
      insights.push({ tipo:'positivo', msg: 'Buen ritmo de ahorro: ' + pctAhorro + '%' });
    if (scoreConsumidor > 70)
      insights.push({ tipo:'identidad', msg: 'Fuiste más consumidor que inversionista en este período' });
    if (logrosCompletados > 0)
      insights.push({ tipo:'positivo', msg: logrosCompletados + ' logro' + (logrosCompletados>1?'s':'') + ' completado' + (logrosCompletados>1?'s':'') });
    if (ingresos === 0 && egresos === 0)
      insights.push({ tipo:'info', msg: 'Sin movimientos registrados en este período' });

    var finGlobal = getFinancieroAvanzado();

    return {
      ok: true,
      tipo: tipo,
      periodo: {
        inicio: Utilities.formatDate(inicio, Session.getScriptTimeZone(), 'dd/MM/yyyy'),
        fin:    Utilities.formatDate(fin,    Session.getScriptTimeZone(), 'dd/MM/yyyy'),
      },
      financiero: { ingresos, egresos: Math.abs(egresos), excedente },
      metricas: finGlobal.metricas || {},
      logros: { completados: logrosCompletados, pendientes: logrosPendientes },
      maslow: { nivelDominante },
      habitos: { total: 0 },
      identidad: { scoreInversionista, scoreConsumidor },
      pctAhorro,
      insights,
    };
  } catch(e) { return { ok: false, error: e.message }; }
}

function enviarSOS(datos) {
  try {
    const ss    = _memoSS();
    const hoja  = _memoSheet('SIMS Relaciones');
    if (!hoja || hoja.getLastRow() < 2) return { ok:false, mensaje:'Sin contactos SOS configurados' };

    const last  = hoja.getLastRow();
    const filaInicioSOS = _getFilaInicio(hoja);
    if (last < filaInicioSOS) return { ok:false, mensaje:'Sin contactos SOS configurados' };
    const data  = hoja.getRange(filaInicioSOS, 1, last-filaInicioSOS+1, 9).getValues();
    const msg   = datos.mensaje || 'Necesito ayuda';
    const timestamp = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'dd/MM/yyyy HH:mm');

    var ubicacion = datos.ubicacion || '';
    var ubicacionTexto = '';
    if (ubicacion) {
      if (ubicacion.indexOf('http') === 0) {
        ubicacionTexto = ubicacion;
      } else {
        var parts = ubicacion.split(',');
        if (parts.length === 2) {
          ubicacionTexto = parts[0].trim() + ', ' + parts[1].trim() + ' → https://maps.google.com/?q=' + parts[0].trim() + ',' + parts[1].trim();
        } else {
          ubicacionTexto = ubicacion;
        }
      }
    }

    let enviados = 0;
    const errores = [];

    for (let i = 0; i < data.length; i++) {
      const sos = String(data[i][8]||'').toLowerCase();
      if (sos !== 'sí' && sos !== 'si') continue;

      const nombre   = String(data[i][0]||'');
      const telefono = String(data[i][6]||'').replace(/[^0-9+]/g,'');
      const email    = String(data[i][7]||'');

      var textoCompleto = 'SOS de RAW Entry - ' + nombre + ' necesita ayuda' +
        '\nFecha: ' + timestamp +
        '\nMensaje: ' + msg +
        (ubicacionTexto ? '\nUbicacion: ' + ubicacionTexto : '\nUbicacion: No disponible');

      if (telefono) {
        try {
          const url = 'https://api.callmebot.com/whatsapp.php?phone=' + telefono + '&text=' + encodeURIComponent(textoCompleto) + '&apikey=YOUR_API_KEY';
          UrlFetchApp.fetch(url);
          enviados++;
        } catch(we) { errores.push('WhatsApp '+nombre+': '+we.message); }
      }
      if (email) {
        try {
          MailApp.sendEmail(email, '🚨 SOS - RAW Entry', textoCompleto.replace(/\*/g,''));
          enviados++;
        } catch(me) { errores.push('Email '+nombre+': '+me.message); }
      }
    }

    let logHoja = _memoSheet('SOS Log');
    if (!logHoja) {
      logHoja = ss.insertSheet('SOS Log');
      logHoja.getRange(1,1,1,3).setValues([['Fecha','Mensaje','Ubicación']]);
    }
    logHoja.appendRow([new Date(), msg, ubicacion]);

    return { ok: true, enviados, errores, mensaje: 'SOS enviado a ' + enviados + ' contacto(s)' };
  } catch(e) { return { ok: false, mensaje: e.message }; }
}

// ══════════════════════════════════════════════════════════
//  getAll — endpoint único
//  v5.006: ahora incluye pensamientos, relaciones, salud, nutricion,
//  entrenamiento y patrimonio para que la Bitácora full-screen y
//  los HUD panels reciban todo en una sola llamada.
// ══════════════════════════════════════════════════════════
function getAll() {
  // v5.008 — MEMOIZACIÓN POR REQUEST + CACHE SERVICE.
  // Dos capas de optimización trabajando juntas:
  //
  // CAPA 1 — Cache Service (entre requests):
  //   Primera llamada del día: calcula todo, guarda resultado en cache
  //   de Apps Script (6h). Llamadas subsiguientes: lee del cache en
  //   ~200ms. Cache se invalida automáticamente desde funciones de
  //   escritura via invalidarCache().
  //
  // CAPA 2 — Memoización por request (dentro de una llamada):
  //   Cuando el cache de Apps Script falla y hay que recalcular, las
  //   sub-funciones (getFijos, getGastos, getNecesidades, getActivityCheck
  //   ...) abrían las mismas hojas múltiples veces (RAW × 5, Pagos × 5,
  //   Activity Check × 4). _memoSheet() guarda los .getValues() en un
  //   mapa volátil durante el request y los reusa. Reduce la primera
  //   llamada del día de ~15s a ~6-8s.
  //
  // v5.007 inicial trajo Capa 1. v5.008 añade Capa 2.
  try {
    var cache = CacheService.getScriptCache();
    var cached = cache.get('lifeos_getAll_v1');
    if (cached) {
      try {
        var parsed = JSON.parse(cached);
        parsed._fromCache = true;
        return parsed;
      } catch(parseErr) {
        cache.remove('lifeos_getAll_v1');
      }
    }

    // v5.008 — Iniciar memoización para este request. Cualquier
    // llamada a _memoSheet() dentro del scope de esta ejecución
    // reusará los .getValues() en vez de re-leer la hoja.
    _memoSheetInit();

    // v5.007 — calcular getFinancieroAvanzado UNA VEZ y pasarlo a
    // getPatrimonio para evitar la doble llamada.
    var finAvz = getFinancieroAvanzado();

    var result = {
      ok:                 true,
      catalogos:          getCatalogos(),
      sheetUrl:           _memoSS().getUrl(),
      fijos:              getFijos(),
      datosMes:           getDatosMes(),
      gastos:             getGastos(),
      logros:             getLogros(),
      necesidades:        getNecesidades(),
      flujoPorMes:        getFlujoPorMes(),
      activityCheck:      getActivityCheck(),
      financieroAvanzado: finAvz,
      apartados:          getApartados(),
      pensamientos:       getPensamientos(),
      relaciones:         getRelaciones(),
      salud:              getSalud(),
      nutricion:          getNutricion(),
      entrenamiento:      getEntrenamiento(),
      patrimonio:         getPatrimonio(finAvz),
    };

    // v5.008 — Limpiar memoización al terminar.
    _memoSheetClear();

    try {
      var serialized = JSON.stringify(result);
      if (serialized.length < 100000) {
        cache.put('lifeos_getAll_v1', serialized, 21600); // 6h
      }
    } catch(cacheErr) {}

    return result;
  } catch(e) {
    _memoSheetClear();
    return { ok:false, error:e.message };
  }
}

// v5.008 — MEMOIZACIÓN POR REQUEST DE REFERENCIAS A HOJAS
// Durante un request (típicamente getAll), evita que cada sub-función
// invoque _memoSS() y getSheetByName(name).
// Cada una de esas llamadas cuesta latencia con Google (~100-200ms
// cada una en frío). Las sub-funciones se llamaban hasta 16 veces,
// re-haciendo las mismas operaciones de "abrir la hoja".
//
// IMPORTANTE: NO memoizamos getValues() porque cada función pide
// rangos distintos. Solo memoizamos las referencias livianas.
// Las sub-funciones siguen funcionando exactamente igual cuando se
// llaman aisladas (fuera de getAll) — _memoSheet detecta si la
// memoización está activa y se comporta acorde.
var _memoCache_ = null;

function _memoSheetInit() {
  _memoCache_ = { ss: null, sheets: {} };
}

function _memoSheetClear() {
  _memoCache_ = null;
}

// Reemplaza _memoSS(). Si la memoización
// está activa, devuelve el Spreadsheet ya obtenido. Si no, llama
// directamente como antes (compatibilidad total).
function _memoSS() {
  if (_memoCache_ === null) {
    return SpreadsheetApp.getActiveSpreadsheet();
  }
  if (_memoCache_.ss === null) {
    _memoCache_.ss = SpreadsheetApp.getActiveSpreadsheet();
  }
  return _memoCache_.ss;
}

// Reemplaza ss.getSheetByName(nombre). Memoiza la referencia a la hoja.
// IMPORTANTE: si la hoja NO existe (resultado null), no lo cacheamos.
// Razón: muchas funciones hacen "if (!hoja) { hoja = ss.insertSheet(...) }"
// — si cacheáramos el null, la siguiente llamada a _memoSheet seguiría
// devolviendo null aunque la hoja ya exista. Esto preserva la semántica
// exacta del código previo.
function _memoSheet(nombre) {
  if (_memoCache_ === null) {
    return SpreadsheetApp.getActiveSpreadsheet().getSheetByName(nombre);
  }
  var cached = _memoCache_.sheets[nombre];
  if (cached !== undefined && cached !== null) {
    return cached;
  }
  var ss = _memoSS();
  var sheet = ss.getSheetByName(nombre);
  if (sheet !== null) {
    _memoCache_.sheets[nombre] = sheet;
  }
  return sheet;
}

// v5.007 — Helper para invalidar el cache de getAll. Se llama desde
// todas las funciones de escritura.
function invalidarCache() {
  try {
    CacheService.getScriptCache().remove('lifeos_getAll_v1');
  } catch(e) {}
}

// ══════════════════════════════════════════════════════════
//  SCORE DE VIDA
// ══════════════════════════════════════════════════════════
function getScoreVida() {
  // v5.008 — Si getScoreVida se llama de forma aislada (no desde
  // getAll), activamos memoización por nosotros mismos. Si ya estaba
  // activa (porque venimos de getAll), respetamos lo que había.
  var weStartedMemo = false;
  if (_memoCache_ === null) {
    _memoSheetInit();
    weStartedMemo = true;
  }
  try {
    const fin     = getFinancieroAvanzado();
    const logros  = getLogros();
    const necs    = getNecesidades();
    const act     = getActivityCheck();
    const sal     = getSalud();
    const rel     = getRelaciones();

    const scores  = {};
    const insights = [];
    const alertas  = [];
    const positivos = [];

    // DINERO (25 pts)
    let scoreDinero = 0;
    const m = (fin.metricas) || {};
    const mes = (fin.mes) || {};
    const pctAhorro = m.porcentajeAhorro || 0;
    const runway = m.runwayDias;
    const velGasto = m.velocidadGasto;

    if (pctAhorro >= 30)      scoreDinero += 10;
    else if (pctAhorro >= 20) scoreDinero += 8;
    else if (pctAhorro >= 10) scoreDinero += 5;
    else if (pctAhorro >= 0)  scoreDinero += 2;

    if (runway === null)    scoreDinero += 5;
    else if (runway >= 90)  scoreDinero += 10;
    else if (runway >= 30)  scoreDinero += 7;
    else if (runway >= 14)  scoreDinero += 4;
    else if (runway >= 7)   scoreDinero += 2;

    if (velGasto === null)   scoreDinero += 3;
    else if (velGasto <= 0)  scoreDinero += 5;
    else if (velGasto <= 10) scoreDinero += 3;
    else if (velGasto <= 25) scoreDinero += 1;

    scores.dinero = Math.min(25, scoreDinero);

    if (pctAhorro < 0)
      alertas.push({ area:'💰', msg:'Estás gastando más de lo que ganas este mes', nivel:'critico' });
    else if (pctAhorro < 10)
      alertas.push({ area:'💰', msg:'Tu capacidad de ahorro es baja — menos del 10%', nivel:'alto' });
    else if (pctAhorro >= 20)
      positivos.push({ area:'💰', msg:'Buen ritmo de ahorro: ' + pctAhorro + '% este mes' });

    if (runway !== null && runway < 7)
      alertas.push({ area:'💰', msg:'Tu runway bajó a ' + runway + ' días — situación crítica', nivel:'critico' });
    else if (runway !== null && runway < 30)
      alertas.push({ area:'💰', msg:'Tu runway es de ' + runway + ' días — cuida tus gastos', nivel:'medio' });

    if (velGasto !== null && velGasto > 30)
      alertas.push({ area:'💰', msg:'Gastas ' + velGasto + '% más rápido en la segunda quincena', nivel:'medio' });

    if (mes.proyeccion && mes.proyeccion.excedente < 0)
      alertas.push({ area:'💰', msg:'Proyección negativa de fin de mes: -$' + Math.abs(mes.proyeccion.excedente).toLocaleString('es-MX'), nivel:'alto' });

    // HÁBITOS (25 pts)
    let scoreHabitos = 0;
    var hojaChecks = _memoSheet('Activity Checks');
    var checksRecientes = 0;
    var checksTotal = 0;
    if (hojaChecks && hojaChecks.getLastRow() > 1) {
      var filaInicioChecks = _getFilaInicio(hojaChecks);
      var lastChecks = hojaChecks.getLastRow();
      var checkData = lastChecks >= filaInicioChecks ? hojaChecks.getRange(filaInicioChecks, 1, lastChecks-filaInicioChecks+1, 3).getValues() : [];
      var hace7 = new Date(); hace7.setDate(hace7.getDate()-7); hace7.setHours(0,0,0,0);
      for (var ci = 0; ci < checkData.length; ci++) {
        var _cn = String(checkData[ci][0]||'').trim();
        if (!_cn || _cn === 'Semana' || _cn.indexOf('FIN EJEMPLOS') >= 0) continue;
        checksTotal++;
        var fCheck = checkData[ci][2];
        if (fCheck instanceof Date && fCheck >= hace7) checksRecientes++;
      }
      if (checksTotal === 0) {
        scoreHabitos = 0;
        insights.push({ area:'⚡', msg:'Sin historial de hábitos — guarda tu primera semana desde Activity Check' });
      } else if (checksRecientes >= 7) {
        scoreHabitos = 25;
        positivos.push({ area:'⚡', msg:'Excelente consistencia de hábitos esta semana: ' + checksRecientes + ' registros' });
      } else if (checksRecientes >= 4) {
        scoreHabitos = 18;
        positivos.push({ area:'⚡', msg:'Buena semana de hábitos: ' + checksRecientes + ' registros' });
      } else if (checksRecientes >= 1) {
        scoreHabitos = 10;
        insights.push({ area:'⚡', msg:'Hábitos inconsistentes esta semana: solo ' + checksRecientes + ' registros' });
      } else {
        scoreHabitos = 2;
        alertas.push({ area:'⚡', msg:'Sin hábitos registrados esta semana', nivel:'medio' });
      }
    } else {
      scoreHabitos = 0;
      insights.push({ area:'⚡', msg:'Sin historial de hábitos — guarda tu primera semana desde Activity Check' });
    }
    scores.habitos = Math.min(25, scoreHabitos);

    // SALUD (20 pts)
    const saludItems = (sal.items) || [];
    const proximas   = (sal.proximas) || [];
    const hoy = new Date(); hoy.setHours(0,0,0,0);
    let scoreSalud = 0;

    if (saludItems.length === 0) {
      insights.push({ area:'🏥', msg:'Sin registros de salud — agrega tus citas y seguimiento' });
    } else {
      scoreSalud = 15;
      const citasVencidas = proximas.filter(c => {
        if (!c.proxima) return false;
        return new Date(c.proxima) < hoy;
      });
      if (citasVencidas.length > 0) {
        scoreSalud -= citasVencidas.length * 3;
        alertas.push({ area:'🏥', msg:'Tienes ' + citasVencidas.length + ' cita(s) médicas vencidas', nivel:'alto' });
      }
      if (proximas.length > 0) {
        const diasProxima = Math.floor((new Date(proximas[0].proxima) - hoy) / 86400000);
        if (diasProxima >= 0 && diasProxima <= 7)
          insights.push({ area:'🏥', msg:'Cita próxima en ' + diasProxima + ' día(s): ' + proximas[0].descripcion });
      }
    }
    scores.salud = Math.max(0, Math.min(20, scoreSalud));

    // RELACIONES (15 pts)
    let scoreRel = 0;
    const personas = (rel.items) || [];

    if (personas.length === 0) {
      scoreRel = 0;
      insights.push({ area:'👥', msg:'Sin relaciones registradas' });
    } else {
      const conEnergia = personas.filter(p => p.energia !== 0);
      const energiaPromedio = conEnergia.length > 0
        ? conEnergia.reduce((s,p) => s + p.energia, 0) / conEnergia.length : 0;

      if (energiaPromedio > 0.3) {
        scoreRel += 8;
        positivos.push({ area:'👥', msg:'Estás rodeado de relaciones que te dan energía' });
      } else if (energiaPromedio < -0.3) {
        scoreRel += 3;
        alertas.push({ area:'👥', msg:'La mayoría de tus relaciones activas te drenan energía', nivel:'medio' });
      } else {
        scoreRel += 5;
      }

      const hace7 = new Date(hoy); hace7.setDate(hoy.getDate() - 7);
      const recientes = personas.filter(p => p.ultimaVez && new Date(p.ultimaVez) >= hace7);
      if (recientes.length === 0) {
        alertas.push({ area:'👥', msg:'No has interactuado con nadie en los últimos 7 días', nivel:'medio' });
        scoreRel += 2;
      } else {
        scoreRel += Math.min(7, recientes.length * 2);
      }
    }
    scores.relaciones = Math.min(15, scoreRel);

    // MENTAL (15 pts)
    const hayLogros = logros && logros.items && logros.items.length > 0;
    let scoreMental = 0;
    if (!hayLogros) {
      insights.push({ area:'🧠', msg:'Sin logros registrados' });
    } else {
      const logrosItems = (logros && logros.items) || [];
      const completados = logrosItems.filter(function(it){ return it.completado==='Sí'||it.completado==='Si'; }).length;
      const total_logros = logrosItems.length;
      const pctLogros = total_logros > 0 ? completados/total_logros : 0;
      if (pctLogros >= 0.75)      scoreMental += 15;
      else if (pctLogros >= 0.5)  scoreMental += 10;
      else if (pctLogros >= 0.25) scoreMental += 5;
      else if (completados > 0)   scoreMental += 2;
      if (completados > 0)
        positivos.push({ area:'🧠', msg:completados + ' logro(s) completado(s) de ' + total_logros });
      else
        insights.push({ area:'🧠', msg:'Tienes ' + total_logros + ' logros pendientes' });
    }
    scores.mental = Math.min(15, scoreMental);

    const total = scores.dinero + scores.habitos + scores.salud + scores.relaciones + scores.mental;
    let estadoGeneral, color;
    if (total >= 85)      { estadoGeneral = 'Excelente — sistema en control'; color = 'ok'; }
    else if (total >= 70) { estadoGeneral = 'Bien — hay áreas de oportunidad'; color = 'ok'; }
    else if (total >= 55) { estadoGeneral = 'Regular — requiere atención'; color = 'warn'; }
    else if (total >= 40) { estadoGeneral = 'Bajo — estás perdiendo control'; color = 'err'; }
    else                  { estadoGeneral = 'Crítico — actúa ahora'; color = 'err'; }

    if (scores.dinero > 15 && scores.mental < 8)
      insights.push({ area:'🧬', msg:'Tus finanzas están bien pero tu claridad mental está baja' });
    if (scores.relaciones < 7 && scores.mental < 8)
      alertas.push({ area:'🧬', msg:'Aislamiento + baja claridad mental — señal de alerta', nivel:'alto' });

    var result = {
      ok: true,
      score: {
        total,
        estado: estadoGeneral,
        color,
        desglose: scores,
        maximos: { dinero:25, habitos:25, salud:20, relaciones:15, mental:15 },
      },
      alertas:   alertas.sort((a,b) => ({critico:0,alto:1,medio:2}[a.nivel]||3) - ({critico:0,alto:1,medio:2}[b.nivel]||3)),
      positivos,
      insights,
      timestamp: Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'dd/MM/yyyy HH:mm'),
      finDetalle: {
        pctAhorro:       m.porcentajeAhorro != null ? m.porcentajeAhorro : null,
        runway:          m.runwayDias != null ? m.runwayDias : null,
        velGasto:        m.velocidadGasto != null ? m.velocidadGasto : null,
        ptsAhorro:       pctAhorro>=30?10:pctAhorro>=20?8:pctAhorro>=10?5:pctAhorro>=0?2:0,
        ptsRunway:       runway===null?5:runway>=90?10:runway>=30?7:runway>=14?4:runway>=7?2:0,
        ptsVel:          velGasto===null?3:velGasto<=0?5:velGasto<=10?3:velGasto<=25?1:0,
        checksRecientes: checksRecientes,
        numSalud:        saludItems.length,
        citasVencidas:   proximas.filter(function(c){ return c.proxima && new Date(c.proxima) < hoy; }).length,
        numPersonas:     personas.length,
        interRecientes:  personas.filter(function(p){ var h7=new Date(hoy); h7.setDate(hoy.getDate()-7); return p.ultimaVez && new Date(p.ultimaVez)>=h7; }).length,
        logrosComp:      (logros&&logros.items||[]).filter(function(it){ return it.completado==='Sí'||it.completado==='Si'; }).length,
        logrosTotal:     (logros&&logros.items||[]).length,
      },
    };
    if (weStartedMemo) _memoSheetClear();
    return result;
  } catch(e) {
    if (weStartedMemo) _memoSheetClear();
    return { ok: false, error: e.message };
  }
}

function getPatrimonio(finAvzPrecalculado) {
  try {
    var ss      = _memoSS();
    var pagos   = _memoSheet('Pagos');
    var lastRow = pagos.getLastRow();

    var fijoData = lastRow < 3 ? [] : pagos.getRange(3, 1, lastRow - 2, 4).getValues();

    var saldoBanco    = 0;
    var saldoFisico   = 0;
    var itemsBanco    = [];
    var itemsFisico   = [];

    var KEYWORDS_BANCO    = ['bbva','santander','banamex','banorte','hsbc','scotiabank','inbursa','nu','nubank','hey','spin','coppel'];
    var KEYWORDS_EFECTIVO = ['beats','efectivo','cash','billetera','wallet'];
    var KEYWORDS_PRESTAMO = ['p','préstamo','prestamo','credito','crédito','deuda'];

    for (var i = 0; i < fijoData.length; i++) {
      var nombre = String(fijoData[i][0] || '').trim();
      if (!nombre) continue;

      var monto  = typeof fijoData[i][1] === 'number' ? fijoData[i][1] : 0;
      var fechaR = fijoData[i][2];
      var tipoD  = String(fijoData[i][3] || '').trim().toLowerCase();

      var fechaStr = '—';
      if (fechaR instanceof Date && !isNaN(fechaR)) {
        fechaStr = Utilities.formatDate(fechaR, Session.getScriptTimeZone(), 'dd/MM/yyyy');
      }

      var tipo;
      var nombreL = nombre.toLowerCase();

      if (tipoD === 'banco' || tipoD === 'bank')                          tipo = 'banco';
      else if (tipoD === 'efectivo' || tipoD === 'cash')                  tipo = 'efectivo';
      else if (tipoD === 'inversión' || tipoD === 'inversion')            tipo = 'inversion';
      else if (tipoD === 'préstamo' || tipoD === 'prestamo' || tipoD === 'deuda') tipo = 'prestamo';
      else if (KEYWORDS_PRESTAMO.indexOf(nombreL) >= 0)                   tipo = 'prestamo';
      else if (KEYWORDS_EFECTIVO.some(function(k){ return nombreL.indexOf(k) >= 0; })) tipo = 'efectivo';
      else if (KEYWORDS_BANCO.some(function(k){ return nombreL.indexOf(k) >= 0; }))    tipo = 'banco';
      else tipo = 'banco';

      if (tipo === 'prestamo') continue;

      if (tipo === 'efectivo') {
        saldoFisico += monto;
        itemsFisico.push({ nombre: nombre, monto: monto, fecha: fechaStr });
      } else {
        saldoBanco += monto;
        itemsBanco.push({ nombre: nombre, monto: monto, fecha: fechaStr });
      }
    }

    var invData  = getInversion();
    var saldoInv = invData.saldo || 0;
    var itemsInv = (invData.items || []).slice(0, 5);

    var total     = saldoBanco + saldoFisico + saldoInv;
    var pctBanco  = total > 0 ? Math.round(saldoBanco  / total * 100) : 0;
    var pctFisico = total > 0 ? Math.round(saldoFisico / total * 100) : 0;
    var pctInv    = total > 0 ? Math.round(saldoInv    / total * 100) : 0;

    var finAvz     = finAvzPrecalculado || getFinancieroAvanzado();  // v5.007: usar el ya calculado si nos lo pasaron
    var gastoDia   = (finAvz.metricas && finAvz.metricas.gastoPorDiaPromedio) || 0;
    var gastoMes   = gastoDia * 30;
    var metaFondo  = Math.round(gastoMes * 3);
    var avanceFondo = metaFondo > 0 ? Math.min(100, Math.round(saldoBanco / metaFondo * 100)) : 0;
    var mesesCubiertos = gastoMes > 0 ? Math.round(saldoBanco / gastoMes * 10) / 10 : 0;
    var saludFondo = mesesCubiertos >= 3 ? 'ok' : mesesCubiertos >= 1 ? 'warn' : 'err';

    return {
      ok:        true,
      total:     total,
      banco:     { saldo: saldoBanco,  pct: pctBanco,  items: itemsBanco  },
      fisico:    { saldo: saldoFisico, pct: pctFisico, items: itemsFisico },
      inversion: { saldo: saldoInv,    pct: pctInv,    rendimientoTotal: invData.rendimientoTotal || 0, items: itemsInv },
      fondo:     { meta: metaFondo, avance: avanceFondo, meses: mesesCubiertos, salud: saludFondo },
    };
  } catch(e) {
    return { ok: false, error: e.message };
  }
}

function getAhorro() {
  try {
    const ss   = _memoSS();
    var hoja = _memoSheet('Ahorro');
    if (!hoja) { hoja = ss.insertSheet('Ahorro'); hoja.getRange(1,1,1,6).setValues([['Fecha','Concepto','Monto','Tipo','Banco','Saldo']]); }
    const last = hoja.getLastRow();
    const filaInicio = _getFilaInicio(hoja);
    if (last < filaInicio) return { ok:true, items:[], saldo:0 };
    const data = hoja.getRange(filaInicio, 1, last-filaInicio+1, 6).getValues();
    var items = []; var saldo = 0;
    for (var i = 0; i < data.length; i++) {
      var r = data[i];
      if (!r[0] && !r[1]) continue;
      var _r0str = String(r[0]||'').trim();
      if (_r0str.indexOf('FIN EJEMPLOS') >= 0 || _r0str === 'Fecha') continue;
      var fechaRaw = r[0];
      var fecha = fechaRaw instanceof Date ? Utilities.formatDate(fechaRaw, Session.getScriptTimeZone(), 'yyyy-MM-dd') : String(fechaRaw||'').slice(0,10);
      var monto = Number(r[2]) || 0;
      saldo = Number(r[5]) || saldo + monto;
      if (!r[1] && !r[2]) continue;
      items.push({ fila:i+2, fecha, concepto:String(r[1]||''), monto, tipo:String(r[3]||''), banco:String(r[4]||''), saldo });
    }
    return { ok:true, items:items.reverse(), saldo };
  } catch(e) { return { ok:false, items:[], saldo:0, error:e.message }; }
}

function guardarAhorro(datos) {
  try {
    invalidarCache(); // v5.007: cualquier escritura tira el cache
    const ss   = _memoSS();
    var hoja   = _memoSheet('Ahorro');
    if (!hoja) { hoja = ss.insertSheet('Ahorro'); hoja.getRange(1,1,1,6).setValues([['Fecha','Concepto','Monto','Tipo','Banco','Saldo']]); }
    var saldoActual = 0;
    var last = hoja.getLastRow();
    if (last > 1) { var s = hoja.getRange(last, 6).getValue(); if (typeof s === 'number') saldoActual = s; }
    var monto = Number(datos.monto) || 0;
    var nuevoSaldo = saldoActual + monto;
    var fila = last + 1;
    var fecha = datos.fecha ? new Date(datos.fecha + 'T12:00:00') : new Date();
    hoja.getRange(fila, 1).setValue(fecha).setNumberFormat('dd/mm/yyyy');
    hoja.getRange(fila, 2).setValue(datos.concepto || '');
    hoja.getRange(fila, 3).setValue(monto);
    hoja.getRange(fila, 4).setValue(datos.tipo || 'Depósito');
    hoja.getRange(fila, 5).setValue(datos.banco || '');
    hoja.getRange(fila, 6).setValue(nuevoSaldo);
    return { ok:true, fila, saldo:nuevoSaldo, mensaje:'Ahorro guardado' };
  } catch(e) { return { ok:false, mensaje:e.message }; }
}

function getNutricion() {
  try {
    var ss   = _memoSS();
    var hoja = _memoSheet('Nutrición');
    if (!hoja) {
      hoja = ss.insertSheet('Nutrición');
      hoja.getRange(1,1,1,11).setValues([['Fecha','Momento','Alimento','Calorías','Proteína (g)','Carbos (g)','Grasa (g)','Sodio (mg)','Fibra (g)','Agua (L)','Notas']]);
    }
    if (hoja.getLastRow() < 2) return { ok:true, dias:{}, semana:[], items:[], hoy:{cal:0,prot:0,carbos:0,grasa:0,agua:0,sodio:0,fibra:0,items:[]} };
    var filaInicio = _getFilaInicio(hoja);
    var lastNut = hoja.getLastRow();
    if (lastNut < filaInicio) return { ok:true, dias:{}, semana:[], items:[], hoy:{cal:0,prot:0,carbos:0,grasa:0,agua:0,sodio:0,fibra:0,items:[]} };
    var data = hoja.getRange(filaInicio, 1, lastNut-filaInicio+1, 11).getValues();
    var hoy = new Date(); hoy.setHours(0,0,0,0);
    var hace7 = new Date(hoy); hace7.setDate(hoy.getDate()-6);
    var hace30 = new Date(hoy); hace30.setDate(hoy.getDate()-30);
    var dias = {};
    var itemsAplanados = []; // ← v5.006: array plano (últimos 30 días) para Bitácora full-screen

    for (var i=0; i<data.length; i++) {
      var r = data[i];
      if (!r[0] && !r[2]) continue;
      var f = r[0] instanceof Date ? new Date(r[0]) : new Date(r[0]);
      if (isNaN(f.getTime())) continue;
      f.setHours(0,0,0,0);
      var fStr = Utilities.formatDate(f, Session.getScriptTimeZone(), 'yyyy-MM-dd');

      // Item enriquecido para .items (incluye fecha)
      var itemEnriquecido = {
        fila:     filaInicio + i,
        fecha:    fStr,
        momento:  String(r[1]||''),
        alimento: String(r[2]||''),
        comida:   String(r[2]||''),
        cal:      Number(r[3])||0,
        calorias: Number(r[3])||0,
        prot:     Number(r[4])||0,
        proteina: Number(r[4])||0,
        carbos:   Number(r[5])||0,
        grasa:    Number(r[6])||0,
        sodio:    Number(r[7])||0,
        fibra:    Number(r[8])||0,
        agua:     Number(r[9])||0,
        notas:    String(r[10]||'')
      };

      // Aplanados — últimos 30 días
      if (f >= hace30) itemsAplanados.push(itemEnriquecido);

      // Sólo agrupar por día los últimos 7 días
      if (f < hace7) continue;

      if (!dias[fStr]) dias[fStr] = { fecha:fStr, cal:0, prot:0, carbos:0, grasa:0, agua:0, sodio:0, fibra:0, items:[] };
      var d = dias[fStr];
      var item = {
        momento:  itemEnriquecido.momento,
        alimento: itemEnriquecido.alimento,
        cal:      itemEnriquecido.cal,
        prot:     itemEnriquecido.prot,
        carbos:   itemEnriquecido.carbos,
        grasa:    itemEnriquecido.grasa,
        sodio:    itemEnriquecido.sodio,
        fibra:    itemEnriquecido.fibra,
        agua:     itemEnriquecido.agua,
        notas:    itemEnriquecido.notas,
        fecha:    fStr
      };
      d.cal+=item.cal; d.prot+=item.prot; d.carbos+=item.carbos; d.grasa+=item.grasa; d.sodio+=item.sodio; d.fibra+=item.fibra;
      if (item.agua>0) d.agua = Math.max(d.agua, item.agua);
      d.items.push(item);
    }

    // Ordenar items aplanados por fecha descendente (más recientes primero)
    itemsAplanados.sort(function(a,b){
      return new Date(b.fecha) - new Date(a.fecha);
    });

    var semana = [];
    for (var dd=0; dd<7; dd++) {
      var dia = new Date(hoy); dia.setDate(hoy.getDate()-dd);
      var dStr = Utilities.formatDate(dia, Session.getScriptTimeZone(), 'yyyy-MM-dd');
      semana.push(dias[dStr] || { fecha:dStr, cal:0, prot:0, carbos:0, grasa:0, agua:0, sodio:0, fibra:0, items:[] });
    }
    var hoyStr = Utilities.formatDate(hoy, Session.getScriptTimeZone(), 'yyyy-MM-dd');
    var dHoy = dias[hoyStr] || { cal:0, prot:0, carbos:0, grasa:0, agua:0, sodio:0, fibra:0, items:[] };

    return { ok:true, dias, semana, items: itemsAplanados, hoy:dHoy };
  } catch(e) {
    return { ok:false, dias:{}, semana:[], items:[], hoy:{cal:0,prot:0,carbos:0,grasa:0,agua:0}, error:e.message };
  }
}

function doTelegram(e) {
  try {
    const body    = JSON.parse(e.postData.contents);
    const msg     = body.message || body.edited_message;
    if (!msg) return ContentService.createTextOutput('ok');
    const chatId  = msg.chat.id;
    const texto   = (msg.text || '').trim().toLowerCase();
    const ALLOWED_CHAT_ID = PropertiesService.getScriptProperties().getProperty('TELEGRAM_CHAT_ID');
    if (ALLOWED_CHAT_ID && String(chatId) !== String(ALLOWED_CHAT_ID)) {
      _tgSend(chatId, '⛔ No autorizado');
      return ContentService.createTextOutput('ok');
    }
    if (texto === '/start' || texto === '/ayuda' || texto === '/help') {
      _tgSend(chatId, _tgMsgAyuda());
    } else if (texto === '/estado' || texto === '/vida') {
      _tgSend(chatId, _tgMsgScore(getScoreVida()));
    } else if (texto === '/hoy') {
      _tgSend(chatId, _tgMsgHoy(getFinancieroAvanzado()));
    } else if (texto === '/semana') {
      _tgSend(chatId, _tgMsgRevision(getRevision('semanal'), 'semanal'));
    } else if (texto === '/mes') {
      _tgSend(chatId, _tgMsgRevision(getRevision('mensual'), 'mensual'));
    } else if (texto === '/runway') {
      const fin = getFinancieroAvanzado();
      const r = fin.metricas ? fin.metricas.runwayDias : null;
      _tgSend(chatId, r !== null ? '🛫 *Runway actual: ' + r + ' días*\nAl ritmo de $' + (fin.metricas.gastoPorDiaPromedio||0).toLocaleString('es-MX') + '/día' : 'Sin datos suficientes');
    } else if (texto === '/proyeccion') {
      const fin = getFinancieroAvanzado();
      const p = fin.mes && fin.mes.proyeccion;
      if (p) {
        _tgSend(chatId, (p.excedente >= 0 ? '✅' : '⚠️') + ' *Proyección fin de mes*\nEgresos: $' + p.egresos.toLocaleString('es-MX') + '\nExcedente: ' + (p.excedente>=0?'+':'') + '$' + p.excedente.toLocaleString('es-MX') + '\nDías restantes: ' + p.diasRestantes);
      } else { _tgSend(chatId, 'Sin datos de proyección'); }
    } else if (texto === '/sos') {
      const r = enviarSOS({ mensaje:'🚨 SOS enviado desde bot de Telegram' });
      _tgSend(chatId, r.ok ? '🚨 SOS enviado a ' + r.enviados + ' contacto(s)' : 'Error: ' + r.mensaje);
    } else {
      _tgSend(chatId, '📊 /estado /hoy /semana /mes /runway /proyeccion /sos');
    }
    return ContentService.createTextOutput('ok');
  } catch(e) {
    Logger.log('doTelegram error: ' + e.message);
    return ContentService.createTextOutput('ok');
  }
}

function _tgSend(chatId, text) {
  var token = PropertiesService.getScriptProperties().getProperty('TELEGRAM_BOT_TOKEN');
  if (!token) { Logger.log('TELEGRAM_BOT_TOKEN no configurado'); return; }
  UrlFetchApp.fetch('https://api.telegram.org/bot' + token + '/sendMessage', {
    method: 'post', contentType: 'application/json',
    payload: JSON.stringify({ chat_id: chatId, text: text, parse_mode: 'Markdown' })
  });
}

    var last = hoja.getLastRow();
    if (last < 2) return { ok: true, mensaje: 'Sin datos', fecha: hoy };

    props.setProperty('ELECTRONICS_RESET_DATE', hoy);
    return { ok: true, mensaje: 'Reset completado', fecha: hoy };
  } catch(e) {
    return { ok: false, mensaje: e.message };
  }
}

// ══════════════════════════════════════════════════════════
//  HELPER — fila inicio datos reales
// ══════════════════════════════════════════════════════════
function _getFilaInicio(hoja) {
  var last = hoja.getLastRow();
  if (last < 2) return 2;
  var colA = hoja.getRange(1, 1, last, 1).getValues();
  for (var i = 0; i < colA.length; i++) {
    var val = String(colA[i][0] || '').trim();
    if (val.indexOf('FIN EJEMPLOS') >= 0 || val.indexOf('=== FIN') === 0 || val.indexOf('--- FIN') === 0) {
      return i + 2;
    }
  }
  return 2;
}

      case 'agregarAActivity':      result = agregarAActivity(body.tipo, body.datos);                   break; // ← v5.006: nueva

// ════════════════════════════════════════════════════════════════════
//
//  HOJA "NOTAS" — crea la hoja y pon estos encabezados en la FILA 1,
//  una columna por celda (A1 a F1):
//
//     A1: ID      B1: Titulo   C1: Cuerpo
//     D1: Color   E1: Slot     F1: Creada
//
//  No pongas nada más en la fila 1. Los datos van de la fila 2 hacia
//  abajo y los gestiona este código automáticamente.
// ════════════════════════════════════════════════════════════════════

function _hojaNotas_() {
  var ss = _memoSS();
  var hoja = _memoSheet('NOTAS');
  if (!hoja) {
    // Si la hoja no existe, la crea con sus encabezados (9 columnas).
    hoja = ss.insertSheet('NOTAS');
    hoja.getRange(1, 1, 1, 9).setValues([[
      'ID','Titulo','Cuerpo','Color','Slot','Creada','X','Y','Conexiones'
    ]]);
  } else {
    // v7.020: si la hoja venía con 6 columnas (sticky notes v6), se le
    // agregan X, Y, Conexiones para el grafo. Migración no destructiva.
    if (hoja.getLastColumn() < 9) {
      hoja.getRange(1, 7, 1, 3).setValues([['X','Y','Conexiones']]);
    }
  }
  return hoja;
}

// Devuelve todas las notas: { notas: [ {id,titulo,cuerpo,color,slot}, ... ] }
function getNotas() {
  try {
    var hoja = _hojaNotas_();
    var ultima = hoja.getLastRow();
    if (ultima < 2) return { notas: [] };
    var datos = hoja.getRange(2, 1, ultima - 1, 9).getValues();
    var notas = [];
    for (var i = 0; i < datos.length; i++) {
      var f = datos[i];
      if (!f[0]) continue; // sin ID → fila vacía, saltar
      // v7.020: conexiones se guarda como lista de IDs separada por comas.
      var conex = String(f[8] || '').split(',')
        .map(function(s){ return s.trim(); })
        .filter(function(s){ return s.length > 0; });
      notas.push({
        id:     String(f[0]),
        titulo: String(f[1] || ''),
        cuerpo: String(f[2] || ''),
        color:  String(f[3] || 'amarillo'),
        slot:   parseInt(f[4], 10) || 0,
        x:      (f[6] === '' || f[6] == null) ? null : Number(f[6]),
        y:      (f[7] === '' || f[7] == null) ? null : Number(f[7]),
        conexiones: conex
      });
    }
    return { notas: notas };
  } catch (err) {
    return { error: err.message, notas: [] };
  }
}

// Inserta una nota nueva. datos = {id,titulo,cuerpo,color,slot,x,y,conexiones}
function guardarNota(datos) {
  try {
    invalidarCache(); // v5.007: cualquier escritura tira el cache
    var hoja = _hojaNotas_();
    var conex = Array.isArray(datos.conexiones)
      ? datos.conexiones.join(',') : (datos.conexiones || '');
    hoja.appendRow([
      datos.id,
      datos.titulo || '',
      datos.cuerpo || '',
      datos.color || 'amarillo',
      (datos.slot != null ? datos.slot : 0),
      new Date(),
      (datos.x != null ? datos.x : ''),
      (datos.y != null ? datos.y : ''),
      conex
    ]);
    return { ok: true };
  } catch (err) {
    return { ok: false, error: err.message };
  }
}

// Localiza la fila (1-indexed) de una nota por su ID. -1 si no existe.
function _filaDeNota_(hoja, id) {
  var ultima = hoja.getLastRow();
  if (ultima < 2) return -1;
  var ids = hoja.getRange(2, 1, ultima - 1, 1).getValues();
  for (var i = 0; i < ids.length; i++) {
    if (String(ids[i][0]) === String(id)) return i + 2;
  }
  return -1;
}

// Actualiza título, cuerpo, color, slot, posición y conexiones.
function actualizarNota(datos) {
  try {
    invalidarCache(); // v5.007: cualquier escritura tira el cache
    var hoja = _hojaNotas_();
    var fila = _filaDeNota_(hoja, datos.id);
    if (fila < 0) {
      // No existía → la crea.
      return guardarNota(datos);
    }
    hoja.getRange(fila, 2).setValue(datos.titulo || '');
    hoja.getRange(fila, 3).setValue(datos.cuerpo || '');
    hoja.getRange(fila, 4).setValue(datos.color || 'amarillo');
    if (datos.slot != null) hoja.getRange(fila, 5).setValue(datos.slot);
    // v7.020: posición en el lienzo del grafo.
    if (datos.x != null) hoja.getRange(fila, 7).setValue(datos.x);
    if (datos.y != null) hoja.getRange(fila, 8).setValue(datos.y);
    // v7.020: conexiones — lista de IDs unida por comas.
    if (datos.conexiones != null) {
      var conex = Array.isArray(datos.conexiones)
        ? datos.conexiones.join(',') : (datos.conexiones || '');
      hoja.getRange(fila, 9).setValue(conex);
    }
    return { ok: true };
  } catch (err) {
    return { ok: false, error: err.message };
  }
}

// Borra una nota por su ID.
function borrarNota(id) {
  try {
    invalidarCache(); // v5.007: cualquier escritura tira el cache
    var hoja = _hojaNotas_();
    var fila = _filaDeNota_(hoja, id);
    if (fila < 0) return { ok: true }; // ya no estaba
    hoja.deleteRow(fila);
    return { ok: true };
  } catch (err) {
    return { ok: false, error: err.message };
  }
}

// Cambia solo el slot (posición en la cuadrícula) de una nota.
function moverNota(id, slot) {
  try {
    invalidarCache(); // v5.007: cualquier escritura tira el cache
    var hoja = _hojaNotas_();
    var fila = _filaDeNota_(hoja, id);
    if (fila < 0) return { ok: false, error: 'Nota no encontrada' };
    hoja.getRange(fila, 5).setValue(parseInt(slot, 10) || 0);
    return { ok: true };
  } catch (err) {
    return { ok: false, error: err.message };
  }
}
